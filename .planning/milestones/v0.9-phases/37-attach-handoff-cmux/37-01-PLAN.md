```js
// @ts-check
import { execFile } from 'node:child_process';
import { loadConfig } from '../config.js';

function getCmuxBinary() {
  return loadConfig().cmux.binary;
}

function run(args, logger) {
  return new Promise((resolve, reject) => {
    logger?.debug('cmux.exec', { cmd: args[0], argc: args.length });
    execFile(getCmuxBinary(), args, { timeout: 15_000 }, (err, stdout, stderr) => {
      if (err) {
        logger?.warn('cmux.fail', { cmd: args[0], stderr: String(stderr || '').slice(0, 200) });
        reject(new Error(`cmux ${args[0]} failed: ${stderr || err.message}`));
        return;
      }
      resolve(stdout.trim());
    });
  });
}
```

Divergences to apply (PATTERNS.md lines 64-73):
1. `binary` comes by arg, not via `getCmuxBinary()` internal lookup.
2. `exec` is injected (no eager import of execFile at top).
3. Args are LITERAL FIXED: `['select-workspace', '--workspace', ref]`.
4. NEVER-THROWS — replace `reject(...)` with `resolve({ok:false, code, detail})`.
   - `!err` → `{ok:true}`
   - `err.code === 'ENOENT'` → `{ok:false, code:'ENOENT', detail: err.message}`
   - `typeof err.code === 'number'` (exit code ≠ 0) → `{ok:false, code:'NON_ZERO_EXIT', detail: err.code}`
   - Sync throw in `exec(...)` invocation → `{ok:false, code:'SPAWN_ERROR', detail: err.message}`
   - Any other err shape → `{ok:false, code:'SPAWN_ERROR', detail: err.message}`
5. Default timeout 5_000 (no 15_000).
