From src/cli/dashboard/focus.js — the TEMPLATE to clone (full module read in read_first):
```javascript
export const FOCUS_VERB = 'select-workspace';
export const FOCUS_FLAG = '--workspace';

/** @typedef {{ ok: true }
 *   | { ok: false, code: 'ENOENT' | 'NON_ZERO_EXIT' | 'SPAWN_ERROR', detail: any }} FocusResult */

export function runFocus({ exec, ref, binary, timeoutMs = 5_000 }) {
  if (typeof exec !== 'function') {
    throw new TypeError('runFocus: `exec` is required (no default — leak guard). ...');
  }
  return new Promise((resolve) => {
    try {
      exec(binary, [FOCUS_VERB, FOCUS_FLAG, ref], { timeout: timeoutMs }, (err, _o, _e) => {
        if (!err) { resolve({ ok: true }); return; }
        if (err.code === 'ENOENT') { resolve({ ok: false, code: 'ENOENT', detail: err.message ?? 'ENOENT' }); return; }
        if (typeof err.code === 'number') { resolve({ ok: false, code: 'NON_ZERO_EXIT', detail: err.code }); return; }
        resolve({ ok: false, code: 'SPAWN_ERROR', detail: err.message ?? String(err) });
      });
    } catch (err) {
      resolve({ ok: false, code: 'SPAWN_ERROR', detail: err instanceof Error ? err.message : String(err) });
    }
  });
}
```
