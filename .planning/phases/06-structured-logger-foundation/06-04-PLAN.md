  ```javascript
  import { describe, it } from 'node:test';
  import assert from 'node:assert/strict';
  import { readFileSync, existsSync } from 'node:fs';
  import { fileURLToPath } from 'node:url';
  import { dirname, resolve, join, relative } from 'node:path';

  const __dirname = dirname(fileURLToPath(import.meta.url));
  const REPO = resolve(__dirname, '..');
  const SRC = join(REPO, 'src');
  const IMPORT_RE = /^\s*import\s+(?:[\s\S]*?from\s+)?['"]([^'"]+)['"]/gm;

  /**
   * Walker transitivo de imports relativos (`./x.js`, `../y.js`).
   * Ignora specifiers bare (`node:fs`, `commander`) — fuera del grafo del proyecto.
   * No sigue dynamic `import()` (el repo no los usa, verificado por grep en RESEARCH A3).
   */
  function walkImports(entry, visited = new Set()) {
    if (visited.has(entry)) return visited;
    if (!existsSync(entry)) return visited;
    visited.add(entry);
    const src = readFileSync(entry, 'utf-8');
    for (const match of src.matchAll(IMPORT_RE)) {
      const spec = match[1];
      if (!spec.startsWith('.')) continue;
      const resolved = resolve(dirname(entry), spec);
      walkImports(resolved, visited);
    }
    return visited;
  }

  describe('LOG-12: vigilante isolation (import-graph)', () => {
    it('src/logger.js exists (sanity: test is meaningful only when the prohibited module is real)', () => {
      assert.equal(existsSync(join(SRC, 'logger.js')), true, 'src/logger.js must exist after Plan 02');
    });

    it('src/logger-noop.js exists and has zero imports', () => {
      const noopPath = join(SRC, 'logger-noop.js');
      assert.equal(existsSync(noopPath), true, 'src/logger-noop.js must exist after Plan 02');
      const src = readFileSync(noopPath, 'utf-8');
      const imports = [...src.matchAll(IMPORT_RE)].map(m => m[1]);
      assert.deepEqual(imports, [], `logger-noop.js must have zero imports, found: ${imports.join(', ')}`);
    });

    it('kodo check does not import src/logger.js transitively', () => {
      const graph = walkImports(join(SRC, 'check.js'));
      // Distinguir logger.js del logger-noop.js: ambos terminan con 'logger' pero sólo uno está prohibido.
      const violators = [...graph].filter(p => /\/logger\.js$/.test(p));
      const relViolators = violators.map(p => relative(REPO, p));
      assert.deepEqual(
        violators,
        [],
        `check.js transitively imports src/logger.js via:\n  ${relViolators.join('\n  ')}\n` +
        `Full graph from check.js:\n  ${[...graph].map(p => relative(REPO, p)).join('\n  ')}`,
      );
    });

    it('logger-noop.js is allowed in the check.js graph (explicit whitelist)', () => {
      // Meta-test: si en algún momento algo en el path de check.js importa logger-noop.js,
      // no debe fallar el test anterior. Esta verificación es defensiva: documenta la
      // distinción clara entre logger.js (prohibido) y logger-noop.js (permitido).
      const graph = walkImports(join(SRC, 'check.js'));
      const noopHits = [...graph].filter(p => p.endsWith('/logger-noop.js'));
      // La aserción no es "debe estar" sino "si está, no rompe nada": el test principal
      // arriba usa un regex que distingue `/logger.js$` de `/logger-noop.js$`.
      for (const hit of noopHits) {
        const src = readFileSync(hit, 'utf-8');
        const imports = [...src.matchAll(IMPORT_RE)].map(m => m[1]);
        assert.deepEqual(imports, [], `logger-noop.js reached from check.js must still have zero imports`);
      }
    });
  });
  ```

  **Puntos clave:**
  - La regex `/\/logger\.js$/` distingue `.../logger.js` de `.../logger-noop.js` sin riesgo de colisión.
  - El mensaje de error incluye el grafo completo relativo al repo → ayuda al debugging cuando falla.
  - `existsSync` sanity check evita falsos negativos (si alguien borra `logger.js` por error, el test deja de ser significativo — mejor fallar explicitamente).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Source files (`src/*.js`) → walker regex | Walker lee source code del propio repo — trust boundary interna; riesgo es analítico (false negatives si alguien introduce dynamic import) |
| `spawnSync` → test de budget | Ejecuta `bin/kodo check` con el env del padre; hereda `$HOME` y `$KODO_*` — puede causar flakiness si otro test no limpió |
| Regex walker vs ESM dynamic `import()` | Walker NO detecta imports dinámicos; si Phase 7+ introduce `await import('./logger.js')` en path de check, el test falla silenciosamente |

## STRIDE Threat Register (foco LOG-12)

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-6-04-01 | Tampering (bypass isolation via dynamic import) | `walkImports` regex no detecta `import()` | accept (v0.3) | RESEARCH A3 verificó que el repo no usa dynamic imports. Mitigación futura: migrar a Opción A (loader tracing) cuando cambie la base |
| T-6-04-02 | Elevation of Privilege | Walker sigue `..` fuera de `src/` | mitigate | El walker solo sigue specifiers que comienzan con `.`; si resuelve fuera de `src/`, no es relevante al test (pero tampoco daña) |
| T-6-04-03 | DoS | Walker entra en ciclo si hay imports circulares | mitigate | `visited` Set corta ciclos (patrón estándar) |
| T-6-04-04 | Tampering (test manipulado con THRESHOLD_MS alto) | Planner/revisor sube el threshold para verde artificial | mitigate | Acceptance criterion caso 3 prohíbe subir el threshold para enmascarar regresión; baseline post-phase está en STARTUP-BASELINE.md (auditable en git history) |
| T-6-04-05 | Information Disclosure | Mensaje de error del walker imprime paths completos | accept | Paths son del repo local (no PII); la info ayuda a debugging |
| T-6-04-06 | Flakiness (CI varianza) | `spawnSync` en CI compartido puede medir 40–80 ms con alta varianza | mitigate | THRESHOLD_MS = `max(50, median*1.15)` incorpora margen; median de 5 runs por test; futuro aumentar a 10 si flaky en CI |
</threat_model>
