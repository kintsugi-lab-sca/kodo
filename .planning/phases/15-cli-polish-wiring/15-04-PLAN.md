       ```javascript
       /**
        * Render the verdict + plane side-effect result in human-readable form.
        * Exhaustive switch over the 4 verdict actions (pass/fail/missing/malformed).
        * D-14: verdict.action coloreado según semántica (pass=green / fail=yellow / missing|malformed=red).
        * D-15: summary block = primeras 3 líneas de result.plane.comment_body (slice, NO re-render).
        *
        * @private
        * @param {any} result
        * @param {(s: string) => void} write
        * @param {import('./format.js').Formatter} fmt
        */
       function renderHuman(result, write, fmt) {
         const { verdict, plane, session } = result;
         write(`Session:      ${session.session_id}\n`);
         write(`Task:         ${session.task_ref}\n\n`);
         write('Verdict:\n');

         // D-14 mapping: verdict.action → 3 colores semánticos.
         const actionColored = (() => {
           switch (verdict.action) {
             case 'pass':      return fmt.green('pass');           // happy path
             case 'fail':      return fmt.yellow('fail');          // soft-fail (recoverable por agente)
             case 'missing':   return fmt.red('missing');          // hard-fail (operador interviene)
             case 'malformed': return fmt.red('malformed');        // hard-fail
             default:          return String(verdict.action);
           }
         })();
         write(`  action:      ${actionColored}\n`);

         // El cuerpo se imprime en color neutro.
         switch (verdict.action) {
           case 'pass':
             write(`  phase_id:    ${verdict.phase_id}\n`);
             write(`  must_haves:  ${verdict.must_haves}\n`);
             break;
           case 'fail':
             write(`  phase_id:    ${verdict.phase_id}\n`);
             write(`  reason:      ${verdict.reason}\n`);
             write(`  detail:      ${verdict.detail}\n`);
             break;
           case 'missing':
             write(`  phase_id:    ${verdict.phase_id}\n`);
             break;
           case 'malformed':
             write(`  phase_id:    ${verdict.phase_id}\n`);
             write(`  detail:      ${verdict.detail}\n`);
             break;
         }
         write('\n');

         // D-15 summary block: slice del result.plane.comment_body (NO re-render).
         // Pitfall #2 Phase 10: NO se llama a renderComment desde aquí — se reusa
         // el string que ya generó verify.js. Determinismo del comentario byte-a-byte intacto.
         if (plane && plane.comment_body) {
           const summaryLines = plane.comment_body.split('\n').slice(0, 3);
           write('Plane comment (summary):\n');
           for (const line of summaryLines) {
             write(`  ${line}\n`);
           }
           write('\n');
         }

         // Línea final: Plane status (existente).
         write(`Plane: commented=${plane.commented} transitioned=${plane.transitioned}\n`);
       }
       ```
