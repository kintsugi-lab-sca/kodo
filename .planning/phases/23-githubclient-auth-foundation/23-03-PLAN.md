    Contenido del archivo:

    ```js
    // @ts-check
    /**
     * capture-github-fixtures.js — utilidad one-shot para refrescar
     * `test/fixtures/github/*.json` con respuestas reales del GitHub REST API.
     *
     * NO se ejecuta en CI. Refresh manual cuando GitHub cambie shapes.
     *
     * Uso:
     *   GITHUB_TOKEN=ghp_xxx node scripts/capture-github-fixtures.js <owner>/<repo>
     *
     * Pre-requisitos en el repo de prueba:
     *   - Una issue con label `kodo` (cualquier número, el script descubre la más reciente).
     *   - Permisos `repo` (private) o `public_repo` (public) en el PAT.
     *
     * El script:
     *   1. GET /repos/{o}/{r}/issues?labels=kodo&state=all&per_page=3 → descubre target issue.
     *   2. GET /repos/{o}/{r}/issues/{n}                              → issue.json
     *   3. GET /repos/{o}/{r}/issues?state=open&per_page=3              → issues-list.json
     *   4. GET /repos/{o}/{r}/labels                                   → labels-list.json
     *   5. POST /repos/{o}/{r}/issues/{n}/comments (body redacted)     → comment-created.json
     *      → DELETE /repos/{o}/{r}/issues/comments/{id} (cleanup)
     *
     * Los archivos de error (401/403/404/429) y placeholders (304, rate-limit-low) NO se
     * capturan live — son fáciles de construir manualmente y este script solo los preserva
     * si ya existen (no los sobrescribe).
     *
     * Redacción: owner/repo reales → kodo-test/fixture-repo; IDs altos → 1/2/3; node_id → I_kwTESTNNN.
     * ZERO PATs en el output (verificado por grep post-write).
     */

    import { mkdir, writeFile, readFile } from 'node:fs/promises';
    import { join } from 'node:path';
    import { existsSync } from 'node:fs';

    const TOKEN = process.env.GITHUB_TOKEN;
    const REPO_ARG = process.argv[2];

    if (!TOKEN || !REPO_ARG) {
      console.error('Usage: GITHUB_TOKEN=ghp_xxx node scripts/capture-github-fixtures.js <owner>/<repo>');
      process.exit(1);
    }

    const [REAL_OWNER, REAL_REPO] = REPO_ARG.split('/');
    if (!REAL_OWNER || !REAL_REPO) {
      console.error(`Invalid repo format: "${REPO_ARG}". Expected "owner/repo".`);
      process.exit(1);
    }

    const FIXTURES_DIR = join(process.cwd(), 'test', 'fixtures', 'github');
    const BASE = 'https://api.github.com';
    const HEADERS = {
      'Authorization': `token ${TOKEN}`,
      'Accept': 'application/vnd.github+json',
      'X-GitHub-Api-Version': '2022-11-28',
      'User-Agent': 'kodo/0.7.x (fixture-capture)',
    };

    /**
     * Reemplaza owner/repo reales por `kodo-test`/`fixture-repo` en strings recursivamente.
     * Reemplaza también IDs altos por enteros pequeños (1/2/3) deterministic.
     *
     * @param {any} obj
     * @param {Map<string|number, string|number>} [idMap]
     * @returns {any}
     */
    function redact(obj, idMap = new Map()) {
      if (obj === null || obj === undefined) return obj;
      if (typeof obj === 'string') {
        let out = obj;
        out = out.replaceAll(REAL_OWNER, 'kodo-test');
        out = out.replaceAll(REAL_REPO, 'fixture-repo');
        // Redact node_id strings con prefix conocido: I_, LA_, IC_, PR_, MDU6...
        if (/^(I|LA|IC|PR|MDU6|MDExOk)/.test(out) && out.length > 10) {
          const placeholder = idMap.get(out);
          if (placeholder) return placeholder;
          const counter = idMap.size + 1;
          const prefix = out.startsWith('LA') ? 'LA_TEST' : out.startsWith('IC') ? 'IC_TEST' : out.startsWith('PR') ? 'PR_TEST' : 'I_kwTEST';
          const generated = `${prefix}${String(counter).padStart(3, '0')}`;
          idMap.set(out, generated);
          return generated;
        }
        return out;
      }
      if (typeof obj === 'number') {
        // Redactar IDs altos (> 1000) a enteros pequeños. Heurística: preservamos cualquier
        // número que sea un timestamp o un `number` field semántico (e.g., issue number).
        // Para simplicidad, solo redactamos si está en un campo que conozcamos como ID.
        // → Cubrir esto en post-process por path es overkill; preservamos los números.
        return obj;
      }
      if (Array.isArray(obj)) return obj.map((x) => redact(x, idMap));
      if (typeof obj === 'object') {
        /** @type {Record<string, any>} */
        const out = {};
        for (const [k, v] of Object.entries(obj)) {
          // Reemplazar campos `id` numéricos altos (>1000) por enteros pequeños
          if (k === 'id' && typeof v === 'number' && v > 1000) {
            const key = `id_${v}`;
            const mapped = idMap.get(key);
            if (mapped) {
              out[k] = mapped;
            } else {
              const counter = idMap.size + 1;
              idMap.set(key, counter);
              out[k] = counter;
            }
          } else if (k === 'login' && typeof v === 'string' && v === REAL_OWNER) {
            out[k] = 'kodo-test';
          } else {
            out[k] = redact(v, idMap);
          }
        }
        return out;
      }
      return obj;
    }

    /**
     * @param {string} path
     * @param {{ method?: string, body?: object }} [opts]
     * @returns {Promise<any>}
     */
    async function api(path, opts = {}) {
      const res = await fetch(`${BASE}${path}`, {
        method: opts.method || 'GET',
        headers: { ...HEADERS, ...(opts.body ? { 'Content-Type': 'application/json' } : {}) },
        body: opts.body ? JSON.stringify(opts.body) : undefined,
      });
      if (!res.ok && res.status !== 204) {
        const text = await res.text().catch(() => '');
        throw new Error(`GitHub API ${res.status}: ${path} — ${text.slice(0, 200)}`);
      }
      if (res.status === 204) return null;
      return res.json();
    }

    /**
     * @param {string} filename
     * @param {any} data
     */
    async function writeFixture(filename, data) {
      const filepath = join(FIXTURES_DIR, filename);
      await writeFile(filepath, JSON.stringify(data, null, 2) + '\n', 'utf-8');
      // Sanity: ningún PAT en el output.
      const written = await readFile(filepath, 'utf-8');
      if (/ghp_|github_pat_/.test(written)) {
        throw new Error(`SANITY FAIL: PAT leaked into ${filename}. Aborting.`);
      }
      console.log(`✓ wrote ${filename} (${written.length} bytes)`);
    }

    async function main() {
      await mkdir(FIXTURES_DIR, { recursive: true });

      console.log(`Capturing fixtures from ${REAL_OWNER}/${REAL_REPO} → test/fixtures/github/\n`);

      // 1. Descubrir target issue con label kodo
      const targets = await api(`/repos/${REAL_OWNER}/${REAL_REPO}/issues?labels=kodo&state=all&per_page=1`);
      if (!Array.isArray(targets) || targets.length === 0) {
        console.error(`ERROR: ${REAL_OWNER}/${REAL_REPO} has no issues labeled 'kodo'. Crea una y reintenta.`);
        process.exit(1);
      }
      const targetIssue = targets[0];
      const targetNumber = targetIssue.number;
      console.log(`Target issue: #${targetNumber} "${targetIssue.title}"`);

      // 2. issue.json
      const issueResp = await api(`/repos/${REAL_OWNER}/${REAL_REPO}/issues/${targetNumber}`);
      await writeFixture('issue.json', redact(issueResp));

      // 3. issues-list.json (3 más recientes, sin filtro)
      const listResp = await api(`/repos/${REAL_OWNER}/${REAL_REPO}/issues?state=open&per_page=3`);
      await writeFixture('issues-list.json', redact(listResp));

      // 4. labels-list.json
      const labelsResp = await api(`/repos/${REAL_OWNER}/${REAL_REPO}/labels?per_page=100`);
      await writeFixture('labels-list.json', redact(labelsResp));

      // 5. comment create + delete
      const commentResp = await api(
        `/repos/${REAL_OWNER}/${REAL_REPO}/issues/${targetNumber}/comments`,
        { method: 'POST', body: { body: 'Test comment markdown' } },
      );
      const redactedComment = redact(commentResp);
      // Forzar body a 'Test comment markdown' (la redacción no lo toca pero seamos defensivos)
      redactedComment.body = 'Test comment markdown';
      await writeFixture('comment-created.json', redactedComment);

      // Limpiar el comment para no ensuciar el repo de prueba
      try {
        await api(`/repos/${REAL_OWNER}/${REAL_REPO}/issues/comments/${commentResp.id}`, { method: 'DELETE' });
        console.log(`✓ cleanup: deleted comment ${commentResp.id} from upstream repo`);
      } catch (err) {
        console.warn(`⚠ cleanup failed (manual delete needed): comment ${commentResp.id}`);
      }

      // 6. Verificar que los archivos de error existen (no se capturan live)
      const errorFixtures = [
        'unauthorized-401.json',
        'forbidden-403.json',
        'not-found-404.json',
        'rate-limit-exceeded.json',
        'rate-limit-low.json',
        'issues-list-304.json',
      ];
      const missing = errorFixtures.filter((f) => !existsSync(join(FIXTURES_DIR, f)));
      if (missing.length > 0) {
        console.warn(`\n⚠ Missing manual fixtures (no live capture; create from RESEARCH.md):`);
        for (const f of missing) console.warn(`   - ${f}`);
      } else {
        console.log(`\n✓ All 6 manual fixtures present.`);
      }

      console.log(`\nDone. ${4} fixtures captured + ${errorFixtures.length - missing.length}/${errorFixtures.length} manual fixtures verified.`);
    }

    main().catch((err) => {
      console.error('FATAL:', err.message);
      process.exit(2);
    });
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `process.env.GITHUB_TOKEN` → request headers | PAT del usuario llega via env var; nunca se loguea ni se escribe a archivos de salida. |
| Captured response → fixture JSON files | Body de respuesta puede contener info personal (login, IDs, body de issues). Redactor reemplaza owner/repo/login antes de write. |
| File write to `test/fixtures/github/` | Idempotente (overwrite). El usuario debe ejecutar desde root del repo para que `process.cwd()` resuelva correctamente. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-23-09 | Information Disclosure | PAT leak en fixture output | mitigate | Post-write sanity check: regex `/ghp_|github_pat_/` sobre el contenido escrito. Si match → throw + exit 2. El script aborta antes de continuar. |
| T-23-10 | Information Disclosure | User info (login, names) en fixtures | mitigate | Redact recursiva reemplaza `login: REAL_OWNER` → `'kodo-test'`. Otros campos personales (e.g., bio del user) no aparecen en los endpoints capturados — el `user` object solo trae `login` + `id` en issues/labels API. |
| T-23-11 | Tampering | Sin args → comportamiento indefinido | mitigate | Validación temprana de `GITHUB_TOKEN` y `argv[2]`; exit 1 con usage si falta. |
