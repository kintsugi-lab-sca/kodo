    /**
     * Captura cada call a process.stdout.write durante fn() y devuelve el array
     * de chunks (string). Restaura el write original con try/finally.
     * @template T
     * @param {() => T | Promise<T>} fn
     * @returns {Promise<{ captured: string[], result: T }>}
     */
    export async function captureStdout(fn) {
      const captured = [];
      const original = process.stdout.write.bind(process.stdout);
      process.stdout.write = (chunk) => { captured.push(String(chunk)); return true; };
      try {
        const result = await fn();
        return { captured, result };
      } finally {
        process.stdout.write = original;
      }
    }
